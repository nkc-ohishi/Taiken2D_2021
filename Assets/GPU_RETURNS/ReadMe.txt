//-------------------------------------------------------------------------------------------------
// Unitypackage for GPU_RETURNS
//-------------------------------------------------------------------------------------------------
Unityを利用したゲーム制作において、共通して利用できると思われるアセット群

//-------------------------------------------------------------------------------------------------
// シーン遷移時のフェードインアウト
//-------------------------------------------------------------------------------------------------
◆使い方
１．GPU_RETURNS - Prefabs - FadeCanvasプレハブを一番初めに再生されるシーンに配置

２．通常のシーン遷移の命令を以下のように変更する
	通常
	SceneManager.LoadScene("SceneName");

	フェードあり（第１引数：シーン名　第２引数：フェード時間）
	FadeManager.Instance.LoadScene("SceneName", 1.0f);
	※ 第２引数にはデフォルト引数(1.0f)が指定してあり省略可

//-------------------------------------------------------------------------------------------------
// BGMを再生する
//-------------------------------------------------------------------------------------------------
◆使い方
１．GPU_RETURNS - Prefabs - BgmManagerプレハブを一番初めに再生されるシーンに配置

２．BGMを再生する命令
	BgmManager.Instance.Play("bgmFileName");

※ 再生したいBGMは GPU_RETURNS - Resources - Audio - BGMフォルダ内に保存する

//-------------------------------------------------------------------------------------------------
// SEを再生する
//-------------------------------------------------------------------------------------------------
◆使い方
１．GPU_RETURNS - Prefabs - SeManagerプレハブを一番初めに再生されるシーンに配置

２．SEを再生する命令
	SeManager.Instance.Play("SeFileName");

※ 再生したいSEは GPU_RETURNS - Resources - Audio - SEフォルダ内に保存する
